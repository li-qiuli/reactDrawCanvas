# react_clip

> Made with create-react-library

## Install

```bash
npm install --save react_clip
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react_clip'
import 'react_clip/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
